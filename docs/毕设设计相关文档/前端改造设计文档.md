## 改造一：报告审核页面支持初审+终审双级审核

### 1.1 问题描述

当前报告审核页面 `report/audit/index.vue` 只有单级审核（通过/驳回），需改造为初审+终审双级审核。

### 1.2 API 层变更

**文件**：`src/api/report.ts`

#### 现有审核相关函数（需保留兼容）

```typescript

export function approveReport(data: any): Promise<void> {
  return post('/report/audit/approve', mapAuditToBackend(data))
}

export function rejectReport(data: any): Promise<void> {
  return post('/report/audit/reject', mapAuditToBackend(data))

// 现有 mapAuditToBackend 函数 - 需修改以支持审核级别
function mapAuditToBackend(data: any): any {
  return {
    ...data,
    approved: data.auditResult === 'pass' || data.auditResult === 'approved',
    auditOpinion: data.auditRemark || data.auditOpinion,
  }

```

#### 新增接口函数

```typescript
export function firstApproveReport(data: any): Promise<void> {
  return post('/report/audit/first-approve', mapAuditToBackend(data))
}

export function firstRejectReport(data: any): Promise<void> {
  return post('/report/audit/first-reject', mapAuditToBackend(data))
}

export function finalApproveReport(data: any): Promise<void> {
  return post('/report/audit/final-approve', mapAuditToBackend(data))
}

export function finalRejectReport(data: any): Promise<void> {
  return post('/report/audit/final-reject', mapAuditToBackend(data))

```

#### 修改 auditReport 函数

```typescript
export function auditReport(data: any): Promise<void> {
  if (data.auditLevel === 1) {
    return data.auditResult === 'pass' ? firstApproveReport(data) : firstRejectReport(data)
  } else if (data.auditLevel === 2) {
    return data.auditResult === 'pass' ? finalApproveReport(data) : finalRejectReport(data)
  }
  // 兼容旧逻辑（无 auditLevel 时走原接口）
  return data.auditResult === 'pass' ? approveReport(data) : rejectReport(data)
}
```

#### 修改 batchAuditReports 函数

```typescript
export function batchAuditReports(data: { reportIds: string[]; auditResult: string; auditRemark: string; auditLevel?: number }): Promise<void> {
  return Promise.all(data.reportIds.map(id =>
    auditReport({ reportId: id, auditResult: data.auditResult, auditRemark: data.auditRemark, auditLevel: data.auditLevel })
  )).then(() => {})
}
```

## 改造二：前端权限过滤修复

### 2.1 问题描述

`permission.ts` 中的 `filterAsyncRoutesAction` 使用 `filter(() => true)` 没有实际过滤效果，所有用户都能看到所有菜单。

### 2.2 改造方案

**文件**：`src/stores/modules/permission.ts`

#### 方案：基于角色权限标识过滤路由

```typescript
async filterAsyncRoutesAction(roles: string[]): Promise<AppRouteRecordRaw[]> {
  const routes = await this.getAsyncRoutesAction()
  if (roles.includes('admin') || roles.includes('SUPER_ADMIN')) {
    return routes
  }
  return this.doFilterRoutes(routes, roles)
},

doFilterRoutes(routes: AppRouteRecordRaw[], roles: string[]): AppRouteRecordRaw[] {
  const result: AppRouteRecordRaw[] = []
  for (const route of routes) {
    const tmp = { ...route }
    if (tmp.children && tmp.children.length > 0) {
      tmp.children = tmp.children.filter((child: AppRouteRecordRaw) => {
        if (!child.meta?.permission) return true
        return this.hasPermission(child.meta.permission, roles)
      })
      if (tmp.children.length > 0) {
        result.push(tmp)
      }
    } else {
      if (!tmp.meta?.permission || this.hasPermission(tmp.meta.permission, roles)) {
        result.push(tmp)
      }
    }
  }
  return result
},

hasPermission(permission: string, roles: string[]): boolean {
  const rolePermissions: Record<string, string[]> = {
    'LAB_DIRECTOR': [
      'specimen:register', 'specimen:receive', 'specimen:list', 'specimen:trace',
      'report:apply', 'report:result', 'report:audit', 'report:query', 'report:critical',
      'equipment:list', 'equipment:monitor', 'equipment:maintenance',
      'statistics:workload', 'statistics:specimen', 'statistics:equipment',
      'ai:analysis', 'ai:records', 'ai:rules',
      'system:user', 'system:role', 'system:menu', 'system:dept', 'system:dict'
    ],
    'LAB_TECHNICIAN': [
      'specimen:register', 'specimen:receive', 'specimen:list', 'specimen:trace',
      'report:apply', 'report:result', 'report:audit',
      'equipment:list', 'equipment:maintenance',
      'ai:analysis', 'ai:records',
      'statistics:workload', 'statistics:specimen'
    ],
    'AUDITOR': [
      'specimen:list', 'specimen:trace',
      'report:audit', 'report:query',
      'statistics:workload', 'statistics:specimen'
    ],
    'DOCTOR': [
      'report:apply', 'report:query', 'report:critical',
      'ai:records',
      'statistics:workload'
    ],
    'NURSE': [
      'specimen:register', 'specimen:list',
      'report:query'
    ]
  }
  for (const role of roles) {
    const perms = rolePermissions[role]
    if (perms && perms.includes(permission)) {
      return true
    }
  }
  return false
}
```

```typescript
import type { App, Directive } from 'vue'
import { useUserStore } from '@/stores/modules/user'

const permissionDirective: Directive = {
  mounted(el, binding) {
    const { value } = binding
    const userStore = useUserStore()
    const permissions = userStore.permissions || []

    if (value && !permissions.includes(value)) {
      el.parentNode?.removeChild(el)
    }
  }
}

export function setupPermissionDirective(app: App) {
  app.directive('permission', permissionDirective)
}
```

**问题**：前端传 `specimenId`，但后端 `SpecimenStorageDTO` 接受 `barcode`（条码）。需要二选一修复：

- **方案A（改前端）**：前端改为传 `barcode`，从 `specimen.barcode` 或 `specimen.specimenNo` 获取
- **方案B（改后端）**：后端 `SpecimenStorageDTO` 同时支持 `specimenId` 和 `barcode`，优先用 `barcode`

建议采用**方案B**，后端 `SpecimenStorageDTO` 新增 `specimenId` 字段，Service 层优先按 `barcode` 查找，若为空则按 `specimenId` 查找。
