## 2 用户管理模块详细设计

用户管理模块由认证服务（lis-auth）和用户服务（lis-user）两个微服务协同完成。认证服务负责登录认证和Token管理，用户服务负责用户、角色、权限、部门等基础数据的CRUD操作。

### 2.1 用户登录认证流程

#### 2.1.1 整体流程描述

用户登录认证采用JWT（JSON Web Token）方案，整体流程分为登录认证、Token签发、请求鉴权和Token刷新四个阶段。

```
用户登录认证时序流程：

客户端                    网关(Gateway)            认证服务(Auth)          用户服务(User)           Redis
  |                          |                        |                      |                     |
  |-- POST /auth/login ----->|-- 转发请求 ------------>|                      |                     |
  |   {username, password}    |                        |                      |                     |
  |                          |                        |-- 查询用户信息 ------>|                     |
  |                          |                        |<-- 返回用户实体 ------|                     |
  |                          |                        |                      |                     |
  |                          |                        |-- BCrypt密码比对 ----|                     |
  |                          |                        |                      |                     |
  |                          |                        |-- 生成access_token --|                     |
  |                          |                        |-- 存储refresh_token->|                     |
  |                          |                        |   (userId->token)    |                     |
  |                          |                        |                      |                     |
  |<-- 返回Token对 ----------|<-- {access_token,      |                     |
  |    refresh_token,         |    refresh_token,      |                     |
  |    userInfo}              |    expiresIn}          |                     |
  |                          |                        |                     |
  |-- GET /specimen/xxx ---->|                        |                     |
  |   Authorization: Bearer   |-- Token校验 ---------->|                     |
  |                          |<-- 用户ID/权限 --------|                     |
  |                          |-- 转发(附加Header) --->|                     |
```

#### 2.1.2 JWT Token生成算法

系统使用双Token机制：access_token用于接口鉴权，有效期2小时；refresh_token用于无感刷新，有效期7天。

**access_token生成逻辑：**

```java
// JWT Payload 结构设计
{
    "sub": "1234567890",          // 用户ID（Subject）
    "username": "zhangsan",       // 用户名
    "roles": ["ADMIN", "DOCTOR"], // 角色编码列表
    "permissions": [              // 权限标识列表
        "sys:user:list",
        "sys:user:add",
        "specimen:register",
        "report:audit"
    ],
    "iat": 1712345678,            // 签发时间（Issued At）
    "exp": 1712352878             // 过期时间（Expiration），当前时间+2小时
}

// JWT生成伪代码
function generateAccessToken(user, roles, permissions):
    claims = new HashMap()
    claims.put("sub", user.id)
    claims.put("username", user.username)
    claims.put("roles", roles.stream().map(r -> r.roleCode).collect(toList()))
    claims.put("permissions", permissions.stream().map(m -> m.permission).collect(toList()))
    claims.put("iat", currentTimeMillis() / 1000)
    claims.put("exp", currentTimeMillis() / 1000 + 7200)  // 2小时有效期

    return Jwts.builder()
        .setClaims(claims)
        .signWith(SignatureAlgorithm.HS512, jwtSecret)  // HS512算法签名
        .compact()
```

**refresh_token生成逻辑：**

```java
// refresh_token Payload 结构（更简洁，仅包含用户标识）
{
    "sub": "1234567890",       // 用户ID
    "type": "refresh",         // Token类型标识
    "iat": 1712345678,
    "exp": 1712950478          // 当前时间+7天
}

// refresh_token存储到Redis
// Key格式：refresh_token:{userId}
// Value：refresh_token字符串
// TTL：7天（604800秒）
redisTemplate.opsForValue().set(
    "refresh_token:" + user.getId(),
    refreshToken,
    7, TimeUnit.DAYS
)
```

#### 2.1.3 JWT验证流程

```java
// JWT验证伪代码
function validateToken(token):
    try:
        // 1. 解析JWT
        claims = Jwts.parserBuilder()
            .setSigningKey(jwtSecret)
            .build()
            .parseClaimsJws(token)
            .getBody()

        // 2. 检查过期时间
        if claims.get("exp") < currentTimeSeconds():
            throw new TokenExpiredException("Token已过期")

        // 3. 提取用户信息
        userId = claims.get("sub")
        username = claims.get("username")
        roles = claims.get("roles")
        permissions = claims.get("permissions")

        // 4. 检查用户是否被禁用（从Redis缓存获取）
        userStatus = redisTemplate.opsForValue().get("user:status:" + userId)
        if userStatus == "0":
            throw new UserDisabledException("账号已被禁用")

        return new JwtUserInfo(userId, username, roles, permissions)

    catch (ExpiredJwtException e):
        throw new TokenExpiredException("Token已过期", e)
    catch (SignatureException e):
        throw new InvalidTokenException("Token签名无效", e)
    catch (MalformedJwtException e):
        throw new InvalidTokenException("Token格式错误", e)
```

#### 2.1.4 Token刷新机制

```java
// Token刷新伪代码
function refreshToken(refreshToken):
    // 1. 验证refresh_token有效性
    claims = parseAndValidate(refreshToken)
    userId = claims.get("sub")
    tokenType = claims.get("type")

    // 2. 校验Token类型
    if tokenType != "refresh":
        throw new InvalidTokenException("非刷新令牌")

    // 3. 从Redis验证refresh_token是否存在（防止Token被主动注销）
    storedToken = redisTemplate.opsForValue().get("refresh_token:" + userId)
    if storedToken == null || storedToken != refreshToken:
        throw new InvalidTokenException("刷新令牌已失效")

    // 4. 查询用户最新信息（角色/权限可能已变更）
    user = userService.getById(userId)
    roles = roleService.getRolesByUserId(userId)
    permissions = menuService.getPermissionsByUserId(userId)

    // 5. 生成新的access_token
    newAccessToken = generateAccessToken(user, roles, permissions)

    // 6. 重新生成refresh_token并更新Redis
    newRefreshToken = generateRefreshToken(user)
    redisTemplate.opsForValue().set(
        "refresh_token:" + userId,
        newRefreshToken,
        7, TimeUnit.DAYS
    )

    return {accessToken: newAccessToken, refreshToken: newRefreshToken}
```

### 2.2 RBAC权限控制实现

#### 2.2.1 角色-菜单-权限三级模型

系统采用经典的RBAC（Role-Based Access Control）模型，通过"用户-角色-菜单-权限"四级关联实现细粒度的权限控制。

```
权限控制模型关系：

sys_user ──────> sys_user_role ──────> sys_role ──────> sys_role_menu ──────> sys_menu
  (用户)              (用户角色表)          (角色)             (角色菜单表)          (菜单)
                                                                                    |
                                                                              permission
                                                                             (权限标识)

菜单类型说明：
  menu_type = 1 (目录)  -> 控制前端菜单显示
  menu_type = 2 (菜单)  -> 控制前端路由访问 + 后端接口权限
  menu_type = 3 (按钮)  -> 控制前端按钮显示 + 后端接口权限
```

#### 2.2.2 权限数据加载流程

```java
// 用户登录时加载权限数据伪代码
function loadUserPermissions(userId):
    // 1. 查询用户关联的角色列表
    roleIds = userRoleMapper.selectRoleIdsByUserId(userId)

    // 2. 查询角色关联的菜单列表（去重）
    menuIds = roleMenuMapper.selectMenuIdsByRoleIds(roleIds)

    // 3. 查询菜单详情并构建树形结构
    allMenus = menuMapper.selectByIds(menuIds)

    // 4. 构建前端路由树（menu_type为1或2的菜单）
    routeMenus = allMenus.stream()
        .filter(m -> m.menuType in [1, 2] && m.status == 1)
        .collect(toList())
    menuTree = buildMenuTree(routeMenus, parentId = 0)

    // 5. 提取权限标识列表（menu_type为2或3的菜单的permission字段）
    permissions = allMenus.stream()
        .filter(m -> m.menuType in [2, 3] && m.status == 1 && m.permission != null)
        .map(m -> m.permission)
        .distinct()
        .collect(toList())

    // 6. 缓存到Redis（避免每次请求都查库）
    redisTemplate.opsForValue().set(
        "user:permissions:" + userId,
        JSON.toJSONString(permissions),
        30, TimeUnit.MINUTES
    )

    return {menuTree, permissions}
```

#### 2.2.3 接口级权限校验

```java
// 自定义权限校验注解
@Target({ElementType.METHOD, ElementType.TYPE})
@Retention(RetentionPolicy.RUNTIME)
public @interface PreAuthorize {
    String value();  // 权限标识，如 "specimen:register"
}

// AOP权限校验切面伪代码
@Aspect
@Component
class PermissionAspect:

    @Around("@annotation(preAuthorize)")
    function doPermissionCheck(joinPoint, preAuthorize):
        // 1. 从请求上下文获取当前用户信息
        currentUser = UserContext.getCurrentUser()
        requiredPermission = preAuthorize.value()

        // 2. 超级管理员直接放行
        if currentUser.hasRole("SUPER_ADMIN"):
            return joinPoint.proceed()

        // 3. 从Redis获取用户权限列表
        permissions = getCachedPermissions(currentUser.getId())

        // 4. 校验是否拥有所需权限
        if !(requiredPermission in permissions):
            throw new ForbiddenException("权限不足，需要: " + requiredPermission)

        return joinPoint.proceed()
```

#### 2.2.4 前端动态路由生成

```javascript
// 前端根据后端返回的菜单树动态生成路由
function generateRoutes(menuTree) {
    const routes = []

    menuTree.forEach(menu => {
        const route = {
            path: menu.path,
            name: menu.component,
            component: () => import(`@/views/${menu.component}.vue`),
            meta: {
                title: menu.menuName,
                icon: menu.icon,
                roles: menu.roles  // 可访问该路由的角色
            }
        }

        // 递归处理子菜单
        if (menu.children && menu.children.length > 0) {
            route.children = generateRoutes(menu.children)
            route.redirect = menu.children[0].path
        }

        routes.push(route)
    })

    return routes
}
```

### 2.3 用户CRUD业务逻辑

#### 2.3.1 用户新增

```java
// 用户新增业务逻辑伪代码
@Transactional
function createUser(userDTO):
    // 1. 参数校验
    if userDTO.username == null || userDTO.username.trim().isEmpty():
        throw new BusinessException("用户名不能为空")
    if userDTO.password == null || userDTO.password.length() < 8:
        throw new BusinessException("密码长度不能少于8位")

    // 2. 用户名唯一性校验
    existUser = userMapper.selectByUsername(userDTO.username)
    if existUser != null:
        throw new BusinessException("用户名已存在")

    // 3. 手机号唯一性校验
    if userDTO.phone != null:
        existPhone = userMapper.selectByPhone(encrypt(userDTO.phone))
        if existPhone != null:
            throw new BusinessException("手机号已被注册")

    // 4. 构建用户实体
    user = new SysUser()
    user.setId(snowflakeIdWorker.nextId())      // 雪花算法生成ID
    user.setUsername(userDTO.username)
    user.setPassword(bCryptPasswordEncoder.encode(userDTO.password))  // BCrypt加密
    user.setRealName(userDTO.realName)
    user.setEmployeeNo(userDTO.employeeNo)
    user.setDeptId(userDTO.deptId)
    user.setPhone(encrypt(userDTO.phone))         // AES加密存储
    user.setEmail(encrypt(userDTO.email))          // AES加密存储
    user.setStatus(1)                              // 默认正常状态
    user.setDeleted(0)

    // 5. 插入用户记录
    userMapper.insert(user)

    // 6. 绑定角色
    if userDTO.roleIds != null && !userDTO.roleIds.isEmpty():
        userRoleList = userDTO.roleIds.stream()
            .map(roleId -> new SysUserRole(user.getId(), roleId))
            .collect(toList())
        userRoleMapper.batchInsert(userRoleList)

    // 7. 清除相关缓存
    redisTemplate.delete("user:permissions:" + user.getId())

    return user.getId()
```

### 2.4 密码加密方案

#### 2.4.1 BCrypt算法原理

BCrypt是一种基于Blowfish密码的密码哈希函数，专为密码存储设计，具有以下特点：

1. **自适应哈希**：通过cost因子控制计算复杂度，可随硬件性能提升而增加计算量；
2. **内置盐值**：每次加密自动生成随机盐值并嵌入哈希结果中，无需单独存储；
3. **抗彩虹表攻击**：由于盐值随机且计算成本可调，有效抵御彩虹表和暴力破解。

**BCrypt哈希结果格式解析：**

```
$2a$10$N.zmdr9k7uOCQb376NoUnuTJ8iAt6Z5EHsM8lE9lBOsl7iAt6Z5EH
|  |  |  |                          |
|  |  |  |                          +-- 哈希值（31字符）
|  |  |  +-- 盐值（22字符）
|  |  +-- Cost因子（10，表示2^10=1024次迭代）
|  +-- 算法版本（2a）
+-- BCrypt标识符
```

#### 2.4.2 密码加密与验证实现

```java
// Spring Security提供的BCryptPasswordEncoder封装
@Component
class PasswordEncoder:

    // BCrypt加密强度配置：cost factor = 10（推荐值10-12）
    encoder = new BCryptPasswordEncoder(10)

    // 加密密码
    function encode(rawPassword):
        return encoder.encode(rawPassword)
        // 每次调用结果不同，因为盐值随机生成
        // 示例：$2a$10$N.zmdr9k7uOCQb376NoUnuTJ8iAt6Z5EHsM8lE9lBOsl7iAt6Z5EH

    // 验证密码
    function matches(rawPassword, encodedPassword):
        return encoder.matches(rawPassword, encodedPassword)
        // 从encodedPassword中提取盐值，对rawPassword进行加密后比对

// 密码修改业务逻辑
function changePassword(userId, oldPassword, newPassword):
    // 1. 查询用户当前密码哈希
    user = userMapper.selectById(userId)
    storedHash = user.getPassword()

    // 2. 验证旧密码
    if !passwordEncoder.matches(oldPassword, storedHash):
        throw new BusinessException("原密码错误")

    // 3. 新密码强度校验
    if newPassword.length() < 8:
        throw new BusinessException("新密码长度不能少于8位")
    if newPassword == oldPassword:
        throw new BusinessException("新密码不能与原密码相同")

    // 4. 加密新密码并更新
    newHash = passwordEncoder.encode(newPassword)
    userMapper.updatePassword(userId, newHash)

    // 5. 使所有Token失效（强制重新登录）
    redisTemplate.delete("refresh_token:" + userId)
```

#### 3.1.1 标本编号生成算法

标本编号是标本的唯一标识，系统采用"日期前缀+流水号"的格式生成，确保全局唯一且便于按日期排序和查询。

```java
// 标本编号格式：SP + yyyyMMdd + 6位流水号
// 示例：SP20260406000001

function generateSpecimenNo():
    // 1. 获取当前日期字符串
    dateStr = LocalDate.now().format(DateTimeFormatter.ofPattern("yyyyMMdd"))
    // dateStr = "20260406"

    // 2. 构建Redis自增Key（按日期维度）
    redisKey = "specimen:no:seq:" + dateStr

    // 3. 使用Redis INCR原子自增获取当日流水号
    sequence = redisTemplate.opsForValue().increment(redisKey)

    // 4. 设置Key过期时间（48小时，防止Redis内存泄漏）
    if sequence == 1:
        redisTemplate.expire(redisKey, 48, TimeUnit.HOURS)

    // 5. 格式化为6位流水号（不足前补零）
    seqStr = String.format("%06d", sequence)

    // 6. 拼接完整编号
    specimenNo = "SP" + dateStr + seqStr
    // specimenNo = "SP20260406000001"

    return specimenNo
```

```java
// 标本批量签收伪代码
@Transactional
function batchReceive(barcodeList, operatorId):
    results = []

    for barcode in barcodeList:
        result = {barcode, success: false, message: ""}

        try:
            // 1. 根据条码查询标本
            specimen = specimenMapper.selectByBarcode(barcode)
            if specimen == null:
                result.message = "标本不存在"
                results.add(result)
                continue

            // 2. 校验标本状态（仅"已采集"状态可签收）
            if specimen.getStatus() != 1:
                result.message = "标本状态不允许签收，当前状态: " + getStatusName(specimen.getStatus())
                results.add(result)
                continue

            // 3. 标本质量检查
            qualityCheck = checkSpecimenQuality(specimen)
            if !qualityCheck.qualified:
                // 不合格标本处理
                specimen.setStatus(5)  // 已退回
                specimenMapper.updateById(specimen)
                addTraceLog(specimen.getId(), "退回", operatorId, qualityCheck.reason)
                result.message = "标本不合格: " + qualityCheck.reason
                results.add(result)
                continue

            // 4. 更新标本状态为"已接收"
            specimen.setStatus(2)
            specimen.setReceiveTime(LocalDateTime.now())
            specimenMapper.updateById(specimen)

            // 5. 记录追溯日志
            addTraceLog(specimen.getId(), "接收", operatorId, "标本签收成功")

            result.success = true
            result.message = "签收成功"
            result.specimenNo = specimen.getSpecimenNo()

        catch (Exception e):
            result.message = "签收异常: " + e.getMessage()

        results.add(result)

    return results  // 返回每个条码的签收结果
```

### 3.3 标本入库排号算法

标本签收后需要分配检测优先级和排号，系统采用多维度优先级排序算法。

```java
// 标本入库排号算法伪代码
function assignTestQueue(specimenId):
    specimen = specimenMapper.selectById(specimenId)

    // 1. 计算标本优先级分数（分数越高优先级越高）
    priorityScore = 0

    // (a) 来源加分：急诊(+50) > 住院(+20) > 门诊(+10) > 体检(+0)
    switch specimen.getSource():
        case "急诊": priorityScore += 50
        case "住院": priorityScore += 20
        case "门诊": priorityScore += 10
        case "体检": priorityScore += 0

    // (b) 等待时间加分：每等待1小时+5分（最高30分）
    waitHours = ChronoUnit.HOURS.between(specimen.getReceiveTime(), LocalDateTime.now())
    timeScore = min(waitHours * 5, 30)
    priorityScore += timeScore

    // (c) 检验项目紧急度加分
    items = specimenItemMapper.selectBySpecimenId(specimenId)
    for item in items:
        if isUrgentItem(item.getItemCode()):
            priorityScore += 40
            break

    // 2. 获取当前队列中的最大排号
    maxQueueNo = specimenMapper.selectMaxQueueNo(specimen.getSpecimenType())
    queueNo = (maxQueueNo != null ? maxQueueNo : 0) + 1

    // 3. 更新标本排号和优先级
    specimen.setQueueNo(queueNo)
    specimen.setPriorityScore(priorityScore)
    specimen.setStatus(3)  // 检测中
    specimenMapper.updateById(specimen)

    // 4. 记录追溯日志
    addTraceLog(specimenId, "排号", currentUserId,
        "排号:" + queueNo + ", 优先级:" + priorityScore)

    return {queueNo, priorityScore}
```

#### 3.4.1 状态流转规则

```java
enum SpecimenStatus:
    REGISTERED(0, "已登记")
    COLLECTED(1, "已采集")
    RECEIVED(2, "已接收")
    TESTING(3, "检测中")

    RETURNED(5, "已退回")

// key: 当前状态, value: 允许转移到的目标状态列表
STATUS_TRANSITIONS = {
    0: [1, 5],       // 已登记 -> 已采集/已退回
    1: [2, 5],       // 已采集 -> 已接收/已退回
    2: [3, 5],       // 已接收 -> 检测中/已退回
    3: [4, 5],       // 检测中 -> 已完成/已退回

function validateStatusTransition(currentStatus, targetStatus):

    if allowedTargets == null || !(targetStatus in allowedTargets):
        throw new BusinessException(
            "标本状态不允许从[" + getStatusName(currentStatus) +
            "]变更为[" + getStatusName(targetStatus) + "]"
        )
    return true

// 状态变更通用方法

    // 2. 更新状态
    oldStatus = specimen.getStatus()
    specimen.setStatus(targetStatus)

    // 3. 根据目标状态更新对应时间字段
    switch targetStatus:
        case 2: specimen.setReceiveTime(LocalDateTime.now())   // 已接收
        case 4: specimen.setCompleteTime(LocalDateTime.now())   // 已完成

    // 4. 记录追溯日志
```

```java
                throw new BusinessException(item.getItemName() + "的结果值超出合理范围")

        // 5. 自动判断异常标志
        abnormalFlag = judgeAbnormal(item.getReferenceRange(), dto.result)

        // 6. 更新报告项目
        item.setResult(dto.result)
        item.setIsAbnormal(abnormalFlag != "N" ? 1 : 0)
        item.setAbnormalFlag(abnormalFlag)
        item.setRemark(dto.remark)
        reportItemMapper.updateById(item)

        // 7. 判断是否为危急值
        if abnormalFlag != "N":
            criticalResult = checkCriticalValue(item.getItemCode(), dto.result)
            if criticalResult.isCritical:
                handleCriticalValue(report, item, criticalResult)

    return "结果录入成功"
```

#### 4.2.2 异常标志判断算法

```java
// 异常标志判断伪代码
// 参考范围格式："3.5-5.5" 或 "<0.5" 或 ">10.0" 或 "阴性/阳性"
function judgeAbnormal(referenceRange, resultValue):
    if referenceRange == null || resultValue == null:
        return "N"

    // 1. 处理数值范围格式 "低值-高值"
    if referenceRange.contains("-") && isNumeric(resultValue):
        parts = referenceRange.split("-")
        low = Double.parseDouble(parts[0].trim())
        high = Double.parseDouble(parts[1].trim())
        value = Double.parseDouble(resultValue)

        if value > high:
            return "H"  // 偏高
        else if value < low:
            return "L"  // 偏低
        else:
            return "N"  // 正常

    // 2. 处理上限格式 ">阈值"
    if referenceRange.startsWith(">") && isNumeric(resultValue):
        threshold = Double.parseDouble(referenceRange.substring(1))
        value = Double.parseDouble(resultValue)
        return value > threshold ? "H" : "N"

    // 3. 处理下限格式 "<阈值"
    if referenceRange.startsWith("<") && isNumeric(resultValue):
        threshold = Double.parseDouble(referenceRange.substring(1))
        value = Double.parseDouble(resultValue)
        return value < threshold ? "L" : "N"

    // 4. 非数值型结果（如阴性/阳性）
    return "N"
```

### 5.2 数据解析算法

#### 5.2.1 ASTM协议数据解析

```java
// ASTM协议数据解析伪代码
// ASTM数据帧格式：
// <STX>[数据段]<ETX>[校验和]<CR><LF>
// 数据段以<CR>分隔，记录以<CRLF>分隔

function parseASTMData(rawData):
    results = []

    // 1. 去除帧头帧尾
    data = rawData.trim()
    if !data.startsWith(String.valueOf((char)0x02)):  // STX
        throw new ParseException("ASTM数据帧缺少STX起始符")
    if !data.contains(String.valueOf((char)0x03)):    // ETX
        throw new ParseException("ASTM数据帧缺少ETX结束符")

    // 2. 提取数据段（STX和ETX之间的内容）
    contentStart = 1
    contentEnd = data.indexOf((char)0x03)
    content = data.substring(contentStart, contentEnd)

    // 3. 校验和验证
    checksumStr = data.substring(contentEnd + 1, contentEnd + 3)
    calculatedChecksum = calculateChecksum(content)
    if Integer.parseInt(checksumStr, 16) != calculatedChecksum:
        throw new ParseException("ASTM校验和错误")

    // 4. 按行分割记录
    records = content.split("\\r")

    for record in records:
        fields = record.split("\\|")

        // 5. 根据记录类型解析
        recordType = fields[0]
        switch recordType:
            case "H":  // 头记录（Header）
                parseHeaderRecord(fields)
            case "P":  // 患者记录（Patient）
                patientInfo = parsePatientRecord(fields)
            case "O":  // 检验订单记录（Order）
                orderInfo = parseOrderRecord(fields)
            case "R":  // 结果记录（Result）
                result = parseResultRecord(fields)
                results.add(result)
            case "L":  // 尾记录（Terminator）
                break  // 结束标志

    return {patientInfo, orderInfo, results}

// 结果记录解析
function parseResultRecord(fields):
    // ASTM结果记录格式：R|序号|项目编码|项目名称|结果值|单位|参考范围|异常标志
    return {
        sequence: fields[1],           // 序号
        itemCode: fields[2],           // 项目编码
        itemName: fields[3],           // 项目名称
        result: fields[4],             // 结果值
        unit: fields[5],               // 单位
        referenceRange: fields[6],     // 参考范围
        abnormalFlag: fields[7]        // 异常标志（H/L/N）
    }
```

#### 5.2.2 校验和计算

```java
// ASTM校验和计算（所有字符的ASCII值异或）
function calculateChecksum(data):
    checksum = 0
    for char in data.toCharArray():
        checksum = checksum ^ (int) char
    return checksum & 0xFF  // 取低8位
```

```java
function aggregateWorkStatistics(startDate, endDate, deptId):
    // 1. 构建查询条件
    queryWrapper = new QueryWrapper<WorkStatistics>()
        .between("stat_date", startDate, endDate)
    if deptId != null:
        queryWrapper.eq("dept_id", deptId)

    // 2. 按日期维度聚合
    dailyStats = workStatisticsMapper.selectList(queryWrapper)

    // 3. 计算汇总指标
    totalSpecimens = dailyStats.stream().mapToInt(s -> s.getSpecimenCount()).sum()
    totalReports = dailyStats.stream().mapToInt(s -> s.getReportCount()).sum()
    avgDailySpecimens = dailyStats.isEmpty() ? 0 : totalSpecimens / dailyStats.size()
    avgDailyReports = dailyStats.isEmpty() ? 0 : totalReports / dailyStats.size()

    // 4. 按科室维度聚合（如果未指定科室）
    if deptId == null:
        deptStats = dailyStats.stream()
            .collect(Collectors.groupingBy(
                WorkStatistics::getDeptName,
                Collectors.summarizingInt(WorkStatistics::getSpecimenCount)
            ))

    // 5. 计算环比增长率
    prevPeriodStart = startDate.minusDays(endDate.compareTo(startDate))
    prevPeriodEnd = startDate.minusDays(1)
    prevTotal = workStatisticsMapper.sumByDateRange(prevPeriodStart, prevPeriodEnd)
    growthRate = prevTotal == 0 ? 0.0 :
        ((double)(totalSpecimens - prevTotal) / prevTotal * 100)

    return {
        totalSpecimens, totalReports,
        avgDailySpecimens, avgDailyReports,
        growthRate,
        dailyStats,
        deptStats
    }
```

#### 8.1.2 条件表达式解析器

```java
// 条件表达式解析器伪代码
// 支持的条件表达式格式：
//   单条件：">100", "<50", "==0", "!=0", ">=3.5", "<=10.0"
//   组合条件：">100 && <200", "<3.5 || >5.5", ">=100 && <=200"
//   多指标条件："WBC>10 && RBC<3.5"

@Component
class ConditionParser:

    // 解析条件表达式并判断是否满足
    function evaluate(expression, context):
        // context: Map<String, Double> 指标编码到值的映射
        // expression: 条件表达式字符串

        // 1. 预处理：去除空格
        expr = expression.trim()

        // 2. 处理逻辑或（||）
        if expr.contains("||"):
            subExprs = expr.split("\\|\\|")
            for sub in subExprs:
                if evaluateSingle(sub.trim(), context):
                    return true
            return false

        // 3. 处理逻辑与（&&）
        if expr.contains("&&"):
            subExprs = expr.split("&&")
            for sub in subExprs:
                if !evaluateSingle(sub.trim(), context):
                    return false
            return true

        // 4. 单条件判断
        return evaluateSingle(expr, context)

    // 单条件判断
    function evaluateSingle(expr, context):
        // 匹配模式：指标名(可选) + 比较运算符 + 数值
        // 示例："WBC>10", ">100", "<=3.5"

        // 提取指标名（如果有的话）
        pattern = "^([A-Za-z_]+)?(>=|<=|!=|==|>|<)(-?\\d+\\.?\\d*)$"
        matcher = Pattern.compile(pattern).matcher(expr)

        if !matcher.find():
            throw new RuleException("无效的条件表达式: " + expr)

        indicator = matcher.group(1)  // 指标名，可能为null
        operator = matcher.group(2)   // 比较运算符
        threshold = Double.parseDouble(matcher.group(3))  // 阈值

        // 获取实际值
        if indicator != null:
            actualValue = context.get(indicator)
            if actualValue == null:
                return false
        else:
            // 无指标名时，使用默认值（单指标规则场景）
            actualValue = context.get("_default")

        // 执行比较
        switch operator:
            case ">":  return actualValue > threshold
            case "<":  return actualValue < threshold
            case ">=": return actualValue >= threshold
            case "<=": return actualValue <= threshold
            case "==": return actualValue == threshold
            case "!=": return actualValue != threshold
```

```java
        rules = ruleMapper.selectEnabled()
        rules.sort((a, b) -> b.getPriority() - a.getPriority())

        // 4. 执行规则匹配
        matchedResults = []
        for rule in rules:
            try:
                if conditionParser.evaluate(rule.getConditionExpr(), context):
                    matchedResults.add({
                        ruleId: rule.getId(),
                        ruleName: rule.getRuleName(),
                        itemCode: rule.getItemCode(),
                        suggestion: rule.getDiagnosisSuggestion(),
                        confidence: rule.getConfidence()
                    })
            catch (Exception e):
                log.warn("规则[{}]执行失败: {}", rule.getRuleName(), e.getMessage())

        // 5. 聚合诊断结果（去除重复建议，计算综合置信度）
        finalResult = aggregateResults(matchedResults)

        // 6. 保存诊断记录
        saveDiagnosisRecord(reportId, report.getPatientId(), finalResult)

        return finalResult

    // 聚合诊断结果
    function aggregateResults(matchedResults):
        if matchedResults.isEmpty():
            return {hasDiagnosis: false, suggestions: [], confidence: 0}

        // 按建议内容去重
        suggestionMap = new LinkedHashMap()
        for result in matchedResults:
            key = result.suggestion
            if !suggestionMap.containsKey(key):
                suggestionMap.put(key, result)

        suggestions = new ArrayList(suggestionMap.values())

        // 计算综合置信度（取所有匹配规则置信度的加权平均）
        avgConfidence = suggestions.stream()
            .mapToDouble(s -> s.confidence)
            .average()
            .orElse(0)

        // 取最高置信度
        maxConfidence = suggestions.stream()
            .mapToDouble(s -> s.confidence)
            .max()
            .orElse(0)

        return {
            hasDiagnosis: true,
            suggestions: suggestions,
            avgConfidence: round(avgConfidence, 2),
            maxConfidence: round(maxConfidence, 2),
            matchCount: suggestions.size()
        }
```

### 8.5 诊断结果置信度计算

```java
// 诊断结果置信度计算伪代码
function calculateConfidence(matchedRules, patientContext):
    /*
    置信度计算综合考虑以下因素：
    1. 规则自身置信度（rule.confidence）- 基础权重
    2. 指标偏离程度 - 偏离参考范围越远，置信度越高
    3. 规则匹配数量 - 多条规则指向同一诊断时，置信度提升
    4. 患者年龄性别因素 - 年龄和性别作为修正因子
    */

    baseConfidence = 0.0

    for rule in matchedRules:
        // 1. 获取规则基础置信度
        ruleConf = rule.getConfidence()

        // 2. 计算指标偏离度修正系数
        itemCode = rule.getItemCode()
        actualValue = patientContext.get(itemCode)
        refRange = getReferenceRange(itemCode, patientContext.gender, patientContext.age)

        // 计算偏离度（偏离参考范围中值的百分比）
        refMid = (refRange.low + refRange.high) / 2
        refRange_width = refRange.high - refRange.low
        deviation = abs(actualValue - refMid) / refRange_width

        // 偏离度修正系数（偏离越大，修正越高，上限+0.1）
        deviationFactor = min(deviation * 0.05, 0.1)

        // 3. 综合置信度
        adjustedConf = min(ruleConf + deviationFactor, 1.0)
        baseConfidence = max(baseConfidence, adjustedConf)

    // 4. 多规则匹配加成（同一诊断方向的多条规则匹配）
    if matchedRules.size() >= 3:
        baseConfidence = min(baseConfidence + 0.05, 1.0)

    // 5. 年龄修正（老年患者指标异常的置信度适当降低）
    if patientContext.age > 65:
        baseConfidence = baseConfidence * 0.95

    return round(baseConfidence, 2)
```

```java
// JWT认证拦截器
@Component
@Slf4j
public class JwtAuthenticationInterceptor implements HandlerInterceptor {

    @Autowired
    private JwtTokenProvider jwtTokenProvider;

    // 白名单路径（不需要Token验证的接口）
    private static final List<String> WHITE_LIST = Arrays.asList(
        "/auth/login",
        "/auth/refresh",
        "/doc.html",
        "/swagger-resources",
        "/v3/api-docs",
        "/favicon.ico"
    );

    @Override
    public boolean preHandle(HttpServletRequest request,
                             HttpServletResponse response,
                             Object handler) throws Exception {
        // 1. OPTIONS预检请求直接放行
        if ("OPTIONS".equalsIgnoreCase(request.getMethod())) {
            return true;
        }

        String requestURI = request.getRequestURI();

        // 2. 白名单路径直接放行
        if isWhiteListed(requestURI):
            return true

        // 3. 提取Token
        String authHeader = request.getHeader("Authorization")
        if authHeader == null || !authHeader.startsWith("Bearer "):
            sendUnauthorizedResponse(response, "未携带Token")
            return false

        String token = authHeader.substring(7)

        // 4. 验证Token
        try:
            jwtUserInfo = jwtTokenProvider.validateToken(token)

            // 5. 将用户信息存入请求上下文（ThreadLocal）
            UserContext.setUserId(jwtUserInfo.getUserId())
            UserContext.setUsername(jwtUserInfo.getUsername())
            UserContext.setRoles(jwtUserInfo.getRoles())
            UserContext.setPermissions(jwtUserInfo.getPermissions())

            return true

        catch (TokenExpiredException e):
            sendUnauthorizedResponse(response, "Token已过期")
            return false
        catch (Exception e):
            sendUnauthorizedResponse(response, "Token无效")
            return false

    @Override
    public void afterCompletion(HttpServletRequest request,
                                HttpServletResponse response,
                                Object handler, Exception ex) {
        // 清除线程上下文，防止内存泄漏
        UserContext.clear()
    }

    private function sendUnauthorizedResponse(response, message):
        response.setStatus(401)
        response.setContentType("application/json;charset=UTF-8")
        response.getWriter().write(
            JSON.toJSONString(Result.fail(401, message))
        )
}
```

```java
        // 1. 获取请求信息
        request = ((ServletRequestAttributes) RequestContextHolder.getRequestAttributes()).getRequest()
        method = point.getSignature().getMethod()
        className = point.getTarget().getClass().getSimpleName()
        methodName = method.getName()
        ip = IpUtil.getIpAddress(request)
        url = request.getRequestURI()
        httpMethod = request.getMethod()

        // 2. 获取请求参数（排除敏感参数）
        params = filterSensitiveParams(getMethodParams(point))

        // 3. 获取当前操作用户
        userId = UserContext.getUserId()
        username = UserContext.getUsername()

        log.info("请求日志 - 用户:{} | {} {} | 类:{}.{} | 参数:{}",
            username, httpMethod, url, className, methodName, params)

        try:
            // 4. 执行目标方法
            result = point.proceed()

            // 5. 记录成功日志
            long elapsed = System.currentTimeMillis() - startTime
            log.info("响应日志 - 用户:{} | {} {} | 耗时:{}ms | 状态:成功",
                username, httpMethod, url, elapsed)

            // 6. 异步保存操作日志到数据库
            saveOperationLog(userId, username, className, methodName,
                url, httpMethod, params, "成功", elapsed, ip)

            return result

        catch (Exception e):
            // 7. 记录失败日志
            long elapsed = System.currentTimeMillis() - startTime
            log.error("异常日志 - 用户:{} | {} {} | 耗时:{}ms | 状态:失败 | 异常:{}",
                username, httpMethod, url, elapsed, e.getMessage())

            saveOperationLog(userId, username, className, methodName,
                url, httpMethod, params, "失败:" + e.getMessage(), elapsed, ip)

            throw e
    }

    // 敏感参数过滤（密码、Token等不记录日志）
    function filterSensitiveParams(params):
        sensitiveKeys = ["password", "token", "idCard", "phone", "email"]
        filteredParams = new HashMap(params)
        for key in sensitiveKeys:
            if filteredParams.containsKey(key):
                filteredParams.put(key, "******")
```

```java
    function <T> T getWithCache(String key, Supplier<T> loader,
                                 long ttl, TimeUnit unit):
        // 1. 尝试从缓存获取
        cached = redisTemplate.opsForValue().get(key)
        if cached != null:
            return (T) cached

        // 2. 缓存未命中，执行加载逻辑
        value = loader.get()

        // 3. 防止缓存穿透：空值也缓存（短时间）
        if value == null:
            redisTemplate.opsForValue().set(key, "", 5, TimeUnit.MINUTES)
            return null

        // 4. 写入缓存
        redisTemplate.opsForValue().set(key, value, ttl, unit)
        return value

    // 批量清除缓存（支持通配符）
    function clearByPattern(String pattern):
        keys = redisTemplate.keys(pattern)
        if keys != null && !keys.isEmpty():
            redisTemplate.delete(keys)
```

### 10.1 路由守卫实现

```javascript
// 路由守卫配置（router/index.ts）
import { createRouter, createWebHistory } from 'vue-router'
import { useUserStore } from '@/stores/modules/user'
import { usePermissionStore } from '@/stores/modules/permission'
import NProgress from 'nprogress'
import 'nprogress/nprogress.css'

// 白名单路由（无需登录即可访问）
const whiteList = ['/login', '/404', '/403']

const router = createRouter({
  history: createWebHistory(),
  routes: [
    {
      path: '/login',
      name: 'Login',
      component: () => import('@/views/login/index.vue'),
      meta: { title: '登录', hidden: true }
    },
    {
      path: '/404',
      name: 'NotFound',
      component: () => import('@/views/error/404.vue'),
      meta: { title: '404', hidden: true }
    },
    {
      path: '/403',
      name: 'Forbidden',
      component: () => import('@/views/error/403.vue'),
      meta: { title: '403', hidden: true }
    }
    // 动态路由在登录后由后端菜单数据生成
  ]
})

// 全局前置守卫
router.beforeEach(async (to, from, next) => {
  NProgress.start()

  const userStore = useUserStore()
  const permissionStore = usePermissionStore()

  // 1. 判断是否有Token
  if (userStore.token) {
    if (to.path === '/login') {
      // 已登录状态访问登录页，重定向到首页
      next({ path: '/' })
      NProgress.done()
    } else {
      // 2. 判断是否已加载用户信息和动态路由
      if (userStore.roles.length > 0) {
        next()  // 已加载，直接放行
      } else {
        try {
          // 3. 获取用户信息（包含角色和权限）
          await userStore.getUserInfo()

          // 4. 根据用户角色生成动态路由
          const dynamicRoutes = await permissionStore.generateRoutes(
            userStore.roles
          )

          // 5. 动态添加路由
          dynamicRoutes.forEach(route => {
            router.addRoute(route)
          })

          // 6. 添加兜底路由（必须在动态路由之后添加）
          router.addRoute({
            path: '/:pathMatch(.*)*',
            redirect: '/404'
          })

          // 7. 重新导航到目标路由（确保动态路由已生效）
          next({ ...to, replace: true })
        } catch (error) {
          // 获取用户信息失败（Token过期或无效），清除状态并跳转登录页
          await userStore.resetState()
          next(`/login?redirect=${to.path}`)
          NProgress.done()
        }
      }
    }
  } else {
    // 无Token
    if (whiteList.includes(to.path)) {
      next()  // 白名单路由直接放行
    } else {
      next(`/login?redirect=${to.path}`)  // 重定向到登录页
      NProgress.done()
    }
  }
})

// 全局后置守卫
router.afterEach((to) => {
  NProgress.done()
  // 设置页面标题
  document.title = to.meta.title
    ? `${to.meta.title} - 实验室管理系统`
    : '实验室管理系统'
})
```

### 10.2 Axios请求拦截器

```javascript
// Axios请求/响应拦截器配置（utils/request.ts）
import axios from 'axios'
import { ElMessage } from 'element-plus'
import { useUserStore } from '@/stores/modules/user'
import router from '@/router'

// 创建Axios实例
const service = axios.create({
  baseURL: '/api',           // API基础路径（通过网关代理）
  timeout: 30000,            // 请求超时时间30秒
  headers: {
    'Content-Type': 'application/json;charset=UTF-8'
  }
})

// 请求拦截器
service.interceptors.request.use(
  config => {
    const userStore = useUserStore()

    // 1. 自动附加Token
    if (userStore.token) {
      config.headers['Authorization'] = `Bearer ${userStore.token}`
    }

    // 2. 生成请求追踪ID
    config.headers['X-Request-Id'] = generateUUID()

    // 3. GET请求添加时间戳防止缓存
    if (config.method === 'get') {
      config.params = {
        ...config.params,
        _t: Date.now()
      }
    }

    return config
  },
  error => {
    console.error('请求错误:', error)
    return Promise.reject(error)
  }
)

// 响应拦截器
service.interceptors.response.use(
  response => {
    const res = response.data

    // 1. 业务状态码判断
    if (res.code !== 200) {
      ElMessage.error(res.message || '请求失败')

      // 2. Token过期处理
      if (res.code === 10003) {
        // Token过期，尝试使用refresh_token刷新
        return handleTokenRefresh(response.config)
      }

      // 3. 权限不足
      if (res.code === 403) {
        router.push('/403')
      }

      return Promise.reject(new Error(res.message || '请求失败'))
    }

    return res
  },
  error => {
    // HTTP状态码处理
    if (error.response) {
      switch (error.response.status) {
        case 401:
          ElMessage.error('登录已过期，请重新登录')
          useUserStore().resetState()
          router.push('/login')
          break
        case 403:
          ElMessage.error('权限不足，拒绝访问')
          router.push('/403')
          break
        case 404:
          ElMessage.error('请求的资源不存在')
          break
        case 500:
          ElMessage.error('服务器内部错误')
          break
        default:
          ElMessage.error(error.response.data?.message || '请求失败')
      }
    } else if (error.code === 'ECONNABORTED') {
      ElMessage.error('请求超时，请稍后重试')
    } else {
      ElMessage.error('网络连接异常')
    }

    return Promise.reject(error)
  }
)

// Token刷新处理（防止并发请求重复刷新）
let isRefreshing = false
let pendingRequests = []

async function handleTokenRefresh(config) {
  if (isRefreshing) {
    // 正在刷新Token，将请求加入等待队列
    return new Promise((resolve) => {
      pendingRequests.push((newToken) => {
        config.headers['Authorization'] = `Bearer ${newToken}`
        resolve(service(config))
      })
    })
  }

  isRefreshing = true
  const userStore = useUserStore()

  try {
    // 调用刷新Token接口
    const res = await axios.post('/api/auth/refresh', {
      refreshToken: userStore.refreshToken
    })

    // 更新Token
    userStore.setToken(res.data.data.accessToken)
    userStore.setRefreshToken(res.data.data.refreshToken)

    // 重新发送等待队列中的请求
    pendingRequests.forEach(cb => cb(res.data.data.accessToken))
    pendingRequests = []

    // 重新发送原始请求
    config.headers['Authorization'] = `Bearer ${res.data.data.accessToken}`
    return service(config)
  } catch (error) {
    // 刷新失败，跳转登录页
    userStore.resetState()
    router.push('/login')
    return Promise.reject(error)
  } finally {
    isRefreshing = false
  }
}

export default service
```
